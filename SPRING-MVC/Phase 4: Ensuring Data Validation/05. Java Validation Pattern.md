```java
    public static ValidationResult validatePassword(String password) {
        if (password == null || password.isEmpty()) {
            return new ValidationResult(false, "Password toh daalo yaar! ğŸ”");
        }
        
        // Detailed password checking
        if (password.length() < 8) {

    public static ValidationResult validateCreditCard(String cardNumber) {
        if (cardNumber == null || cardNumber.trim().isEmpty()) {
            return new ValidationResult(false, "Card number daalo bhai! ğŸ’³");
        }
        
        cardNumber = cardNumber.replaceAll("\\s+", "").replaceAll("-", "");
        
        // Check if it's all digits

        // Check card type

            cardType = "Mastercard";
        } else if (amexPattern.matcher(cardNumber).matches()) {

        } else {
            return new ValidationResult(false, "Unknown card type! ğŸ¤”");
        }
        
        return new ValidationResult(true, "Valid " + cardType + " card! ğŸ’");
    }
    
    // Luhn Algorithm implementation
    private static boolean isValidLuhn(String cardNumber) {
        int sum = 0;
        boolean alternate = false;
        
        for (int i = cardNumber.length() - 1; i >= 0; i--) {
            int digit = Character.getNumericValue(cardNumber.charAt(i));
            
            if (alternate) {
                digit *= 2;
                if (digit > 9) {
                    digit = (digit % 10) + 1;
                }
            }
            
            sum += digit;
            alternate = !alternate;
        }
        
        return (sum % 10 == 0);
    }

    public static ValidationResult validateURL(String url) {
        if (url == null || url.trim().isEmpty()) {
            return new ValidationResult(false, "URL missing hai yaar! ğŸŒ");

        String urlPattern = "^(https?|ftp)://[^\\s/$.?#].[^\\s]*$";

        } else {
            return new ValidationResult(false, "Invalid URL format! http:// or https:// lagao! ğŸ”§");
        }
    }

        if (pattern.matcher(date).matches()) {
            return new ValidationResult(true, "Valid date format! ğŸ“†");
        } else {
            return new ValidationResult(false, "Date format: DD/MM/YYYY or DD-MM-YYYY hona chahiye! ğŸ—“ï¸");
        }
    }
```

## ğŸ“‹ Rules/Guidelines

### âœ… Golden Rules for Validation

#### 1. âœ… Always Compile Patterns Once
```java

// âŒ Wrong - Compiling every time (slow!)

    return Pattern.compile("email_regex").matcher(email).matches();
}
```

#### 2. âœ… Null Check Pehle Karo
```java
// âœ… Correct

    if (input == null || input.trim().isEmpty()) {
        return false;
    }
    // validation logic...
}

// âŒ Wrong - NullPointerException milega!

    return input.matches("some_regex"); // Boom! ğŸ’¥
}
```

#### 3. âœ… Sanitize Input Before Validation
```java
// âœ… Correct
phone = phone.replaceAll("\\s+", "").replaceAll("-", ""); // Remove spaces and dashes
email = email.trim().toLowerCase(); // Clean and normalize

// âŒ Wrong - Direct validation without cleaning
Pattern.matches(PHONE_PATTERN, " 98765 - 43210 "); // False negative!
```

#### 4. âœ… Use Meaningful Error Messages
```java
// âœ… Correct

// âŒ Wrong

```

#### 5. âœ… Escape Special Characters
```java

    // IFSC code validation
    private static final Pattern IFSC_CODE = Pattern.compile("^[A-Z]{4}0[A-Z0-9]{6}$");
    
    // PAN card validation
    private static final Pattern PAN_CARD = Pattern.compile("^[A-Z]{5}[0-9]{4}[A-Z]{1}$");
    
    // Aadhaar number validation
    private static final Pattern AADHAAR = Pattern.compile("^[2-9]{1}[0-9]{3}[0-9]{4}[0-9]{4}$");
    
    public static class BankAccount {
        private String accountNumber;
        private String ifscCode;
        private String panNumber;
        private String aadhaarNumber;
        private String email;
        private String phone;
        
        public BankAccount(String accountNumber, String ifscCode, String panNumber, 
                          String aadhaarNumber, String email, String phone) {
            // Validate karte hain sab kuch!
            validateAccountDetails(accountNumber, ifscCode, panNumber, aadhaarNumber, email, phone);
            
            this.accountNumber = accountNumber;
            this.ifscCode = ifscCode.toUpperCase();
            this.panNumber = panNumber.toUpperCase();
            this.aadhaarNumber = aadhaarNumber;
            this.email = email.toLowerCase();
            this.phone = phone;
        }
        
        private void validateAccountDetails(String accountNumber, String ifscCode, 
                                          String panNumber, String aadhaarNumber,

            if (!ACCOUNT_NUMBER.matcher(accountNumber).matches()) {
                throw new IllegalArgumentException("Invalid account number! 9-18 digits hone chahiye ğŸ¦");
            }
            
            // IFSC validation
            if (!IFSC_CODE.matcher(ifscCode).matches()) {
                throw new IllegalArgumentException("Invalid IFSC code! Format: ABCD0123456 ğŸ›ï¸");
            }
            
            // PAN validation
            if (!PAN_CARD.matcher(panNumber).matches()) {
                throw new IllegalArgumentException("Invalid PAN! Format: ABCDE1234F ğŸ“„");
            }
            
            // Aadhaar validation
            if (!AADHAAR.matcher(aadhaarNumber).matches()) {

            }
            
            // Email validation
            if (!ValidationPatterns.validateEmail(email).isValid()) {
                throw new IllegalArgumentException("Invalid email address! ğŸ“§");
            }
            
            // Phone validation
            if (!ValidationPatterns.validatePhoneNumber(phone, "india").isValid()) {
                throw new IllegalArgumentException("Invalid phone number! ğŸ“±");
            }
            
            System.out.println("âœ… All banking validations passed! Account created successfully! ğŸ‰");
```

```java
        private void validateProduct(String sku, String name, double price, String category) {
            // SKU validation
            if (!PRODUCT_SKU.matcher(sku).matches()) {
                throw new IllegalArgumentException("Invalid SKU! Format: ABC-1234-XY ğŸ“¦");
            }
            
            // Name validation
            if (name == null || name.trim().length() < 3) {

            }
            
            // Price validation

            }
            
            // Category validation
            String[] validCategories = {"ELECTRONICS", "CLOTHING", "BOOKS", "HOME", "SPORTS"};
            boolean validCategory = false;

                if (cat.equals(category.toUpperCase())) {

                    break;
                }
            }
            
            if (!validCategory) {
                throw new IllegalArgumentException("Invalid category! Valid: ELECTRONICS, CLOTHING, BOOKS, HOME, SPORTS ğŸ“‚");
            }
            
            System.out.println("âœ… Product validation successful! ğŸ¯");

        private void validateOrder(String orderId, String customerEmail, String shippingAddress, String discountCode) {
            // Order ID validation (ORD-timestamp format)
            if (!orderId.matches("^ORD-\\d{10,13}$")) {
                throw new IllegalArgumentException("Invalid Order ID! Format: ORD-1234567890 ğŸ“‹");

            }
            
            // Discount code validation (optional)
```
